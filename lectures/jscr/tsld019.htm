    function Act(){


    var Text = document.exampleForm.userText.value;

    var ind = document.exampleForm.pickList.selectedIndex;

    if( ind == 0 ) return;

    else if ( ind == 1 ) showMessageBox( Text );

    else if ( ind == 2 ) showWindow( Text );

    else if ( ind == 3 ) showStatusBar( Text );

    document.exampleForm.pickList.selectedIndex = 0;

    }
